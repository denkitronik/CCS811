148  printf("The sensor resistance measurement has reached or exceeded the maximum range.\r\n");
149  error = 1;
150  }
151  if (appReg.error_id.meas_mode_invalid) {
152  printf("Request to write an unsupported mode to MEAS_MODE.\r\n");
153  error = 1;
154  }
155  if (appReg.error_id.read_reg_invalid) {
156  printf("Read request to a mailbox ID that is invalid.\r\n");
157  error = 1;
158  }
159  if (appReg.error_id.write_reg_invalid) {
160  printf("Write request to a mailbox ID that is invalid.\r\n");
161  error = 1;
162  }
163  if (!error) {
164  printf("No errors happened!\r\n");
165  }
166  printf("............................\r\n", message);
167  appReg = appRegAux;
168 }

175 void readRegister(const char reg, int numBytes) {
176  selectRegister(reg);
177  readI2C(numBytes);
178  if (getRegister(ALG_RESULT_DATA) == getRegister(reg, 0))
179  if (appReg.alg_result_data.eco2 > 255) {
180  appReg.alg_result_data.eco2 = ((appReg.alg_result_data.eco2 >> 8) | (appReg.alg_result_data.eco2 << 8));
181  if (appReg.alg_result_data.eco2 > 32768) appReg.alg_result_data.eco2 = appReg.alg_result_data.eco2 - 32768;
182  }
183  if (appReg.alg_result_data.tvoc > 255) {
184  appReg.alg_result_data.tvoc = ((appReg.alg_result_data.tvoc >> 8) | (appReg.alg_result_data.tvoc << 8));
185  if (appReg.alg_result_data.tvoc > 32768) appReg.alg_result_data.tvoc = appReg.alg_result_data.tvoc - 32768;
186  }
187 }

242 void connectI2C(const char address) {
243  char *filename = "/dev/i2c-1";
244  if ((file = open(filename, O_RDWR)) < 0) {
245  perror("Failed to open the i2c bus");
246  exit(1);
247  }
248  if (ioctl(file, I2C_SLAVE, address) < 0) {
249  printf("Failed to acquire bus access and/or talk to slave.\r\n");
250  /* ERROR HANDLING; you can check errno to see what went wrong */
251  exit(1);
252  }
253  readRegister(HW_ID);
254  if (appReg.hw_id == 0x81) {
255  writeRegister(STATUS);
256  writeRegister(APP_START);
257  } else {
258  printf("Failed: HW_ID = 0x81 does not match with this device.\r\n");
259  }
260 }

344 void writeI2CBytes(const char reg, int numBytes) {
345  unsigned char bufferData[9] = {reg, 0, 0, 0, 0, 0, 0, 0, 0};
346  for (char i = 0; i < numBytes; i++) bufferData[i + 1] = appReg.buffer[i];
347  if (write(file, bufferData, (numBytes + 1)) != (numBytes + 1)) {
348  /* ERROR HANDLING: i2c transaction failed */
349  DEBUG("Writing to the I2C bus failed\r\n");
350  } else {
351  DEBUG("writeI2CBytes: \t\t");

353  DEBUG("count=%d\r\n", numBytes + 1);
354  }
355 }
